## Solution

### 带备忘录迭代

```C++
class Solution {
public:
    int f[50] = {0,1,1,2,0};
    int size = 3;
    int fib(int n) {
        if(n>size){
            for(;size<n;size++){
                f[size+1] = f[size]+f[size-1];
            }
        }
        return f[n];
    }
};
```

### 记录之前状态

```C++
class Solution {
public:
    vector<int> dp = {0, 1,1};
    int fib(int n) {
        if(n < 2){return n;}
        int f1  = 0, f2 = 1, t;

        while(n-- > 1){
            // 当前值
            t = f1+f2;
            // 更新
            f1 = f2;
            f2 = t;
        }
        return f2;
    }
};
```
