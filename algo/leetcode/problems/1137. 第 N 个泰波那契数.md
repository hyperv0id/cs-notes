## Solution
```C++
class Solution {
public:
    vector<int> dp = {0,1,1};
    int tribonacci(int n) {
        int siz = dp.size();
        while(siz <= n){
            dp.push_back(dp[siz-1] + dp[siz-2] + dp[siz-3]);
            siz++;
        }
        return dp[n];
    }
};
```
